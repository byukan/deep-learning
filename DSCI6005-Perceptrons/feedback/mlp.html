class MultiLayerPerceptron(object):
    def __init__(self, NB_FEATURE, NB_HIDDEN, NB_CLASS):
        self.W1 = np.random.randn(NB_FEATURE, NB_HIDDEN)
        self.b1 = np.random.randn(NB_HIDDEN)
        self.W2 = np.random.randn(NB_HIDDEN, NB_CLASS)
        self.b2 = np.random.randn(NB_CLASS)        
        
    def softmax(self, S):
        return np.exp(S) / np.sum(np.exp(S), axis=1, keepdims=True)
    
    def f(self, X):
        X1 = self.softmax(np.dot(X,self.W1) + self.b1)
        X2 = self.softmax(np.dot(X1,self.W2) + self.b2)
        return X2
    
    
    def predict(self, X):        
        pred = self.f(X)                       
        return pred
    
    def evaluate(self, X, y):    
        return np.mean((self.predict(X)).argmax(axis=1) == y)

class MultiLayerPerceptronWithRandomOptimizer(MultiLayerPerceptron):
    
    def __init__(self, NB_FEATURE, NB_HIDDEN, NB_CLASS):
        MultiLayerPerceptron.__init__(self, NB_FEATURE, NB_HIDDEN, NB_CLASS)
        
    def _forward(self, X, y):
        Y = np.zeros([len(X), NB_CLASS])        
        Y[np.arange(len(X)), y] = 1
        P = self.predict(X)
        return -np.mean(np.array([np.sum(y * np.log(p)) for p, y in zip(P, Y)]))
    
    def fit(self, X, y, nb_epoch=1):
        for _ in range(nb_epoch):
            curr_W1, curr_b1, curr_W2, curr_b2 = self.W1, self.b1, self.W2, self.b2  
            curr_loss = self._forward(X, y)  
            
            self.W1 = np.random.randn(self.W1.shape[0], self.W1.shape[1])
            self.b1 = np.random.randn(self.b1.shape[0])
            self.W2 = np.random.randn(self.W2.shape[0], self.W2.shape[1])
            self.b2 = np.random.randn(self.b2.shape[0])
                                                                                                                                
            if curr_loss < self._forward(X, y):                                     
                self.W1, self.b1, self.W2, self.b2 = curr_W1, curr_b1, curr_W2, curr_b2
